# useQueue

Just like `useState`, you can pass a default value (e.g. `useQueue([1, 2, 3])`)

```js
import { useQueue } from 'use-data-structures'

function App() {
  const input = useRef()
  const queue = useQueue()

  const addQueue = () => {
    queue.enqueue(input.current.value)
    input.current.value = null
  }

  const removeQueue = () => {
    queue.dequeue()
  }

  const peekQueue = () => {
    alert(queue.peek())
  }

  return (
    <div>
      <p>Queue size: {queue.size}</p>
      <div>
        <input ref={input} placeholder="value to queue" />{' '}
        <button onClick={addQueue}>Queue</button>{' '}
        <button onClick={removeQueue}>Dequeue</button>{' '}
        <button onClick={peekQueue}>Peek</button>
      </div>
    </div>
  )
}
```
